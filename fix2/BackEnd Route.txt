const multer = require('multer');
const path = require('path');

// Storage cho custom field files
const customFieldStorage = multer.diskStorage({
  destination: (req, file, cb) => {
    // Kiểm tra xem có phải custom field file không
    if (file.fieldname.startsWith('customFieldFile_')) {
      cb(null, path.resolve(config.upload.customField || config.upload.employee));
    } else {
      cb(null, path.resolve(config.upload.employee));
    }
  },
  filename: (req, file, cb) => {
    const uniqueSuffix = Date.now() + '-' + Math.round(Math.random() * 1E9);
    cb(null, uniqueSuffix + '-' + file.originalname);
  }
});

const uploadCustomFieldFiles = multer({
  storage: customFieldStorage,
  limits: { fileSize: 20 * 1024 * 1024 },
  fileFilter: (req, file, cb) => {
    // Accept tất cả files bắt đầu với customFieldFile_
    if (file.fieldname.startsWith('customFieldFile_')) {
      cb(null, true);
    } else {
      cb(null, false);
    }
  }
});

// ✅ Route
router.post(
  '/',
  authenticate,
  setCustomCreateUploadPath,
  uploadImage.single('avatar'),
  checkImageValidationError,
  uploadCustomFieldFiles.any(), // ✅ Accept tất cả files
  parseCustomField,
  validate(employeeValidation.createEmployee),
  employeeController.create,
);
