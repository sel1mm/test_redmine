} else if (key === 'customField') {
  const customFieldObj = values.customField || {};

  const customFieldArray = [];

  Object.entries(customFieldObj).forEach(([name, value]) => {
    if (value === undefined || value === null || value === '') return;

    // ðŸ“Œ DATE
    if (moment.isMoment(value)) {
      customFieldArray.push({
        name,
        value: value.format('YYYY-MM-DD'),
      });
      return;
    }

    // ðŸ“Œ BOOLEAN
    if (typeof value === 'boolean') {
      customFieldArray.push({
        name,
        value: value ? '1' : '0',
      });
      return;
    }

    // ðŸ“Œ NUMBER
    if (typeof value === 'number') {
      customFieldArray.push({
        name,
        value: String(value),
      });
      return;
    }

    // ðŸ“Œ FILE (UPLOAD GIá»NG AVATAR)
    if (Array.isArray(value) && value[0]?.originFileObj) {
      const file = value[0].originFileObj;

      // append file riÃªng
      formData.append(`customFieldFiles`, file);

      // placeholder â€“ backend sáº½ map láº¡i
      customFieldArray.push({
        name,
        value: file.name, // backend sáº½ replace báº±ng path
      });
      return;
    }

    // ðŸ“Œ STRING
    customFieldArray.push({
      name,
      value: String(value),
    });
  });

  if (customFieldArray.length > 0) {
    formData.append('customField', JSON.stringify(customFieldArray));
  }
}
