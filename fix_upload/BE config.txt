module.exports = {
  upload: {
    employee: './uploads/employee',
    customField: './uploads/customField', // ✅ Thêm path cho custom field
  }
};


const storage = multer.diskStorage({
  destination: (req, file, cb) => {
    let uploadPath;
    
    if (file.fieldname === 'avatar') {
      uploadPath = path.resolve(config.upload.employee);
    } else if (file.fieldname.startsWith('customFieldFile_')) {
      uploadPath = path.resolve(config.upload.customField || config.upload.employee);
    } else {
      uploadPath = path.resolve(config.upload.employee);
    }
    
    // Đảm bảo folder tồn tại
    if (!fs.existsSync(uploadPath)) {
      fs.mkdirSync(uploadPath, { recursive: true });
    }
    
    cb(null, uploadPath);
  },
  filename: (req, file, cb) => {
    const uniqueSuffix = Date.now() + '-' + Math.round(Math.random() * 1E9);
    cb(null, uniqueSuffix + '-' + file.originalname);
  }
});

const uploadImage = multer({
  storage,
  fileFilter,
  limits: { fileSize: 20 * 1024 * 1024, files: 10, fields: 30, parts: 40 },
});
